#  >> brew tap hashicorp/tap
#  >> brew install hashicorp/tap/terraform

# >> terraform init
# >> gcloud auth login
# >> terraform validate
# >> terraform plan
# >> terraform apply
# >> gcloud auth revoke

# if you want to destroy the terraform
# terraform destroy
